\begin{algorithm}[!ht]
\caption{\textproc{dotProduct}}
\label{alg:vectordotproduct}
	\begin{algorithmic}[1]
		\Require $\mathit{left}, \mathit{right}$
		\State \textbf{return} $\left(\mathit{left.x} \cdot \mathit{right.x}\right) + \left(\mathit{left.y} \cdot \mathit{right.y}\right)$
	\end{algorithmic}
\end{algorithm}
