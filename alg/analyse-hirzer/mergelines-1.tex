\begin{algorithm}[!ht]
\caption{\textproc{mergeLines} (Ãœberblick)}
\label{alg:mergelines1}
\begin{algorithmic}[1]
	\Require $L$
	\State $\mathit{distancepool} \gets \infty$
	\Cost{$c_{1}$}{$1$}
	\label{alg:mergelines1-distancepool}
	\For{$i \gets 0$ \textbf{to} $i <$ \Call{getLineCount}{$L$}}
	\Cost{$c_{2}$}{$\bigl(1 + \Theta(1)\bigr)(l + 1)$}
	\label{alg:mergelines1-loop-start}
		\State \Call{freePool}{$\mathit{distancepool}$}
		\Cost{$c_{3}$}{$(l\cdot \Theta(1)$}
		\label{alg:mergelines1-cleardistancepool}
		\State $\mathit{start} \gets$ \Call{getLineSegment}{$L,i$}
		\Cost{$c_{4}$}{$(l \cdot \bigl(1 + \Theta(1)\bigr)$}
		\label{alg:mergelines1-getline}
		\State \ldots \Comment Suche zwei kompatible Liniensegmente
		\If{$\lnot$ \Call{getDistanceCount}{$\mathit{distancepool}$}}
		\Cost{$c_{6}$}{$(l \cdot \bigl(1 + \Theta(1)\bigr)$}
		\label{alg:mergelines1-hasdistance-start}
			\State \textbf{continue}
			\Cost{$c_{7}$}{$l$}
		\EndIf
		\label{alg:mergelines1-hasdistance-end}
		\State \ldots \Comment Verbinde Liniensegmente mit kurzem Abstand zuerst
		\label{alg:mergelines1-mergelines}
		\State $i \gets i + 1$
		\Cost{$c_{10}$}{$l$}
	\EndFor
	\label{alg:mergelines1-loop-end}
\end{algorithmic}
\end{algorithm}
