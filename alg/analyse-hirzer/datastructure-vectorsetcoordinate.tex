\begin{algorithm}[!ht]\small
\caption{\textproc{vectorSetCoordinate}}
\label{alg:vectorsetcoordinate}
	\begin{algorithmic}[1]
		\Require $\mathit{vector}, x, y$
		\State $\mathit{vector.x} \gets x$
		\Cost{$c_{1}$}{$2$}
		\State $\mathit{vector.y} \gets y$
		\Cost{$c_{2}$}{$2$}
	\end{algorithmic}
\end{algorithm}
