\begin{algorithm}[!ht]\small
\caption{\textproc{getDistanceCount}}
\label{alg:distancepool-distancecount}
\begin{algorithmic}[1]
	\Require $p$
	\If{$\lnot p$}
	\Cost{$c_{1}$}{$1$}
		\State \textbf{return}
		\Cost{$c_{2}$}{$1$}
	\EndIf
	\State \textbf{return} $\mathit{p.count}$
	\Cost{$c_{4}$}{$2$}
\end{algorithmic}
\end{algorithm}
