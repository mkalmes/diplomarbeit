\begin{algorithm}[ht]
\caption{\textproc{squaredLength}}
\label{alg:vectorsquaredlength}
	\begin{algorithmic}[1]
		\Require $\mathit{vector}$
		\State \textbf{return} $\left(\mathit{vector.x} \cdot \mathit{vector.x}\right) + \left(\mathit{vector.y}
		 \cdot \mathit{vector.y}\right)$
		\Cost{$c_{1}$}{$7$}
	\end{algorithmic}
\end{algorithm}
