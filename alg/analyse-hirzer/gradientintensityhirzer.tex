\begin{algorithm}[ht]
\caption{\textproc{gradientIntensity} nach Hirzer}
\label{alg:gradientintensityhirzer}
\begin{algorithmic}[1]
	\Require $I, x, y, w, h$
	\State $p_{x_1} \gets$ \Call{getRGBValue}{$I, x - 1, y - 1,\mathit{green}, w, h$}
	\label{alg:gradientintensityhirzer-readstart}
	\State $p_{x_2} \gets$ \Call{getRGBValue}{$I, x    , y - 1,\mathit{green}, w, h$}
	\State $p_{x_3} \gets$ \Call{getRGBValue}{$I, x + 1, y - 1,\mathit{green}, w, h$}
	\State $p_{x_4} \gets$ \Call{getRGBValue}{$I, x - 1, y + 1,\mathit{green}, w, h$}
	\State $p_{x_5} \gets$ \Call{getRGBValue}{$I, x    , y + 1,\mathit{green}, w, h$}
	\State $p_{x_6} \gets$ \Call{getRGBValue}{$I, x + 1, y + 1,\mathit{green}, w, h$}
	\State $p_{y_1} \gets$ \Call{getRGBValue}{$I, x - 1, y - 1,\mathit{green}, w, h$}
	\State $p_{y_2} \gets$ \Call{getRGBValue}{$I, x - 1, y    ,\mathit{green}, w, h$}
	\State $p_{y_3} \gets$ \Call{getRGBValue}{$I, x - 1, y + 1,\mathit{green}, w, h$}
	\State $p_{y_4} \gets$ \Call{getRGBValue}{$I, x + 1, y - 1,\mathit{green}, w, h$}
	\State $p_{y_5} \gets$ \Call{getRGBValue}{$I, x + 1, y    ,\mathit{green}, w, h$}
	\State $p_{y_6} \gets$ \Call{getRGBValue}{$I, x + 1, y + 1,\mathit{green}, w, h$}
	\label{alg:gradientintensityhirzer-readend}
	\State $g_x \gets 0$
	\State $g_y \gets 0$
	\State $g_x \gets g_x + p_{x_1}$
	\label{alg:gradientintensityhirzer-convolutestart}
	\State $g_x \gets g_x + \left(p_{x_2} \cdot 2\right)$
	\State $g_x \gets g_x + p_{x_3}$
	\State $g_x \gets g_x - p_{x_4}$
	\State $g_x \gets g_x - \left(p_{x_5} \cdot 2\right)$
	\State $g_x \gets g_x - p_{x_6}$
	\State $g_y \gets g_y + p_{y_1}$
	\State $g_y \gets g_y + \left(p_{y_2} \cdot 2\right)$
	\State $g_y \gets g_y + p_{y_3}$
	\State $g_y \gets g_y - p_{y_4}$
	\State $g_y \gets g_y - \left(p_{y_5} \cdot 2\right)$
	\State $g_y \gets g_y - p_{y_6}$
	\label{alg:gradientintensityhirzer-convoluteend}
	\State $\mathit{slope} \gets \infty$
	\State \Call{vectorSetCoordinate}{$\mathit{slope},\mathit{gy},\mathit{gx}$}
	\label{alg:gradientintensityhirzer-vector-start}
	\State \Call{normalized}{$\mathit{slope}$}
	\label{alg:gradientintensityhirzer-vector-end}
	\State \textbf{return} $\mathit{slope}$
\end{algorithmic}
\end{algorithm}
