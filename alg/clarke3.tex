\begin{algorithm}[!ht]
	\caption{RANSAC}
	\label{src:ransac}
	\begin{algorithmic}[1]
	\Require $edgels>0$
	\Procedure{RANSAC}{$edgels$}
		\While{$edgels\not=0 \lor i<maxIteration$}
			\For{1--25}
				\State 2 zufällige \glspl{edgel} auswählen und Kriterien prüfen
				\State Anzahl Support-\glspl{edgel} bestimmen
			\EndFor
			\If{Support-\glspl{edgel} > Schwellwert}
				\State Linie erkannt
				\State Support-Edgels aus Eingabemenge entfernen
			\EndIf
			\State $i \gets i + 1$
		\EndWhile
	\EndProcedure
	\end{algorithmic}
\end{algorithm}
