\begin{algorithm}[ht]
\caption{Edgel lesen}
\label{alg:edgelpool-getedgel}
\begin{algorithmic}[1]
	\Require $p,i$
	\If{$!p$}
		\State \textbf{return}
	\EndIf
	\State $c \gets p(\mathit{count})$
	\If{$! c > i$}
		\State \textbf{return}
	\EndIf
	\State \textbf{return} $p(\mathit{data}[i])$
\end{algorithmic}
\end{algorithm}
